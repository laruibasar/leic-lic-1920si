6 /*
7  * A classe SerialEmitter envia para os varios modulos de hardware modules
8  * os dados que recebe do controlo de software.
9  *
10  * O mapeamento para o Output Port da placa vai ser mapeado de acordo com o seguinte
11  * esquema de acordo com os bits disponiveis:
12  * 7 6 5 4 3 2 1 0
13  * In : 0 BSY 0 0 0 0 0 0
14  * Out : 0 0 0 0 DOOR LCD SCLK SDX
15  *
16  *
17  * Este arranjo permite-nos, apenas com o hardware atual, ter ainda disponivel mais
18  * 4 enderecamentos disponiveis para modulos de hardware.
19  *
20  * Do Input Port da placa verificamos o sinal de Busy do hardware para fazer uma
21  * espera do envio dos dados em linha
22  */
23

27 public class SerialEmitter {
28  public enum Destination {LCD, DOOR_MECHANISM};
29 
30  private final static int MASK_DATA = 0x01;
31  private final static int MASK_SCLK = 0x02;
32  private final static int MASK_LCD = 0x04;
33  private final static int MASK_DOOR = 0x08;
34  private final static int MASK_BUSY = 0x40;
35  private final static int TX_FRAME_SIZE = 5;
36 
37  private static int write_addr;
38  private static int parity;
39 
40  // inicia a classe
41  public static void init() {
42  HAL.init();
43  write_addr = 0;
44  parity = 0;
45  HAL.setBits(MASK_LCD);
46  HAL.setBits(MASK_DOOR);
47  }
48 
49  // envia uma trama para o SerialReceiver identificando o destino em addr
50  // e os bits de dados em 'data'
51  // o método faz o envio, sem validar se está busy, para isso esta disponivel
52  // o método isBusy que deve ser utilizado pelas classes antes de fazer a chamada
53  public static void send(Destination addr, int data) {
54  switch (addr) {
55  case LCD:
56  write_addr = MASK_LCD;
57  break;
58  case DOOR_MECHANISM:
59  write_addr = MASK_DOOR;
60  break;
61  default:
62  return;
63  }
64  HAL.clrBits(write_addr);
65 
66  // simples verificacao de controlo da transmissao
67  // dependente do valor da transmissao de par ou impar
68  parity = 0;
69  for (int i = 0; i < TX_FRAME_SIZE; i++) {
70  HAL.clrBits(MASK_SCLK);
71  int tx_data = MASK_DATA & data;
72  parity += (tx_data);
73  HAL.writeBits(MASK_DATA, tx_data);
74  data >>= 1;
75  HAL.setBits(MASK_SCLK);
76  }
77 
78  // enviar a paridade do sinal
79  HAL.clrBits(MASK_SCLK);
80  parity %= 2;
81  HAL.writeBits(MASK_DATA, parity);
82  HAL.setBits(MASK_SCLK);
83 
84  HAL.clrBits(MASK_SCLK);
85  HAL.setBits(MASK_SCLK);
86 
87  HAL.clrBits(MASK_DATA);
88  HAL.clrBits(MASK_SCLK);
89  HAL.setBits(write_addr);
90  }
91 
92  // retorna true se o canal série estiver ocupado
93  // se ocupado, valor 1
94  public static boolean isBusy() {
95  return HAL.isBit(MASK_BUSY);
96  }
