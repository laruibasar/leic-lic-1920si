106  // leitura de um numero, controlando a apresentacao do input do utilizador
107  private static int readInteger(int ndigits, boolean obsfucate, char obs, long timeout, int lin, int col) {
108  int value = 0;
109  int i = 0;
110  showInputRequest(ndigits, obsfucate, obs, lin, col);
111 
112  do {
113  char c = KBD.waitKey(timeout);
114 
115  switch (c) {
116  case 0:
117  value = ETIMEOUT;
118  i = ndigits;
119  break;
120  case '*':
121  if (i > 0) {
122  value = 0;
123  i = 0;
124  showInputRequest(ndigits, obsfucate, obs, lin, col);
125  } else {
126  value = EABORT;
127  i = ndigits;
128  }
129  break;
130  case '#':
131  continue;
132  default:
133  value = (value * 10) + c - '0';
134  showCharacter(c, lin, col + i);
135  i++;
136  break;
137  }
138  } while (i < ndigits);
139 
140  return value;
141  }

143  private static void showInputRequest(int ndigits, boolean obsfucate, char obs, int lin, int col) {
144  if (obsfucate) {
145  String msg = "";
146  for (int i = 0; i < ndigits; i++) {
147  msg = msg + obs;
148  }
149  showMessage(msg, lin, col);
150  LCD.cursor(lin, col);
151  cursorSet(CURSOR_SET);
152  }
153  }
