21 // Controla o estado do mecanismo de abertura da porta
22 public class DoorMechanism {
23  private static final int MASK_DATA_SIZE = 0xf;
24  private static final int DOOR_OPEN = 1;
25  private static final int DOOR_CLOSE = 0;
26 
27  // Inicia a classe, estabelecendo os valores iniciais
28  public static void init() {

31  }

33  // Envia comando para abrir a porta, com o parametro de velocidade
34  // O sinal de dados para o SerialEmitter vai ser:
35  // 0 0 0 (V3) (V2) (V1) (V0) (OC) de acordo com a especificao projeto
36  public static void open(int velocity) {
37  send(((MASK_DATA_SIZE & velocity) << 1) | DOOR_OPEN);
38  }
39 
40  // Envia comando para fechar a porta, com o parameto de velocidade
41  public static void close(int velocity) {
42  send( ((MASK_DATA_SIZE & velocity) << 1) | DOOR_CLOSE);
43  while (!finished());
44  }

52  // Verifica se o comando anterior est√° concluido
